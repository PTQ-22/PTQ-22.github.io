Powtarzającego się znaku

Algorytm powtarzającego się znaku - algorytm kompresji bezstratnej, w którym wielokrotnie powtarzające się znaki zastępujemy ich ilością poprzedzoną wybranym znakiem specjalnym, który nie występuje w kompresowanym pliku. Np:

AAAAAAA => ~7A

void compress_repeat_char(string filename)
{
    ifstream input_file;
    ofstream output_file;
    input_file.open(filename);
    output_file.open("compressed.txt");

    const char SPECIAL_CHAR = '~';

    char prev, curr;
    int counter = 1;
    input_file >> prev;

    while (input_file >> curr) {
        if (curr == prev) {
            counter++;
        }
        else {
            if (counter > 3) {
                output_file << SPECIAL_CHAR << counter << prev;
            }
            else {
                for (int i = 0; i < counter; ++i) {
                    output_file << prev;
                }
            }
            counter = 1;
        }
        prev = curr;
    }
    for (int i = 0; i < counter; ++i) {
        output_file << prev;
    }

    input_file.close();
    output_file.close();
}
