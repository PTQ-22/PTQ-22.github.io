#include <bits/stdc++.h>
using namespace std;

pair<int, int> start_pos;
vector<vector<bool>> visited;
vector<string> grid;
int width;
int height;
const vector<pair<int, int>> MOVES = {
    {0, 1}, {1, 0}, {-1, 0}, {0, -1}
};

bool dfs(pair<int, int> v)
{
    visited[v.first][v.second] = true;
    if (v.first == 0 || v.second == 0 || v.first == height - 1 || v.second == width - 1)
    {
        grid[v.first][v.second] = '.';
        return true;
    }
    bool isOnPath = false;
    for (auto move : MOVES) {
        int y = v.first + move.first;
        int x = v.second + move.second;
        if (grid[y][x] != 'X' && !visited[y][x]) {
            isOnPath |= dfs({y, x});
        }
    }
    if (isOnPath) grid[v.first][v.second] = '.';
    return isOnPath;
}

int main()
{
    fstream file;
    file.open("labirynt.txt");

    string line;
    int i = 0;
    while (getline(file, line))
    {
        grid.push_back(line);
        for (int j = 0; j < line.size(); ++j) {
            if (line[j] == '@') {
                start_pos = {i, j};
            }
        }
        i++;
    }

    height = grid.size();
    width = grid[0].size();

    visited.resize(height, vector<bool>(width, false));

    dfs(start_pos);
    grid[start_pos.first][start_pos.second] = '@';

    cout << "EXIT FROM MAZE: \n\n";
    for (auto x : grid) {
        cout << x << '\n';
    }

    file.close();
    return 0;
}
