Lista dwukierunkowa

Lista dwukierunkowa - lista, która różni się od jednokierunkowej tym, że każdy jej element posiada adres nie tylko do elementu za nim, ale również do elementu przed nim. Pozwala to na łatwiejsze wykonywanie niektórych operacji np. usuwania elementów z środka listy

struct Person {
    string name, surname;
    int age;
};

struct Node {
    Node* prev;
    Person person;
    Node* next;
};

Node* first = nullptr;
Node* last = nullptr;
int list_size = 0;

void push_front(Person person)
{
    Node *tmp = new Node;
    tmp->next = first;
    tmp->prev = nullptr;
    tmp->person = person;

    if (first == nullptr) {
        last = tmp;
        first = tmp;
    }
    else {
        first->prev = tmp;
        first = tmp;
    }
    list_size++;
}

void push_back(Person person)
{
    Node *tmp = new Node;
    tmp->next = nullptr;
    tmp->prev = last;
    tmp->person = person;

    if (first == nullptr) {
        first = tmp;
        last = tmp;
    }
    else {
        last->next = tmp;
        last = tmp;
    }
    list_size++;
}

Person pop_front()
{
    Person res = first->person;
    first = first->next;
    first->prev = nullptr;
    list_size--;
    return res;
}

Person pop_back()
{
    Person res = last->person;
    last = last->prev;
    last->next = nullptr;
    list_size--;
    return res;
}

Person pop(int index)
{
    Node* ptr = first;

    for (int i = 0; i < index; ++i) {
        ptr = ptr->next;
        if (ptr == nullptr)
            return {"-1", "-1", -1};
    }

    Person res = ptr->person;

    ptr->prev->next = ptr->next;
    ptr->next->prev = ptr->prev;

    list_size--;
    return res;
}

void push_behind(Person person, int index)
{
    Node* ptr = first;

    for (int i = 0; i < index; ++i) {
        ptr = ptr->next;
        if (ptr == nullptr)
            return;
    }
    Node *tmp = new Node;
    tmp->person = person;
    tmp->prev = ptr;
    tmp->next = ptr->next;

    ptr->next->prev = tmp;
    ptr->next = tmp;
    list_size++;
}

void push_before(Person person, int index)
{
    Node* ptr = first;

    for (int i = 0; i < index; ++i) {
        ptr = ptr->next;
        if (ptr == nullptr)
            return;
    }
    Node *tmp = new Node;
    tmp->person = person;
    tmp->prev = ptr->prev;
    tmp->next = ptr;

    ptr->prev->next = tmp;
    ptr->prev = tmp;
    list_size++;
}

void printList()
{
    Node* ptr = first;
    int counter = 0;
    while (ptr != nullptr) {
        Person p = ptr->person;
        cout << counter++ << ' ';
        cout << p.name << ' ' << p.surname << ' ' << p.age << '\n';
        ptr = ptr->next;
    }
}

void printReversedList()
{
    Node* ptr = last;
    int counter = list_size - 1;
    while (ptr != nullptr) {
        Person p = ptr->person;
        cout << counter-- << ' ';
        cout << p.name << ' ' << p.surname << ' ' << p.age << '\n';
        ptr = ptr->prev;
    }
}
