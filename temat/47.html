Pole wielokąta - iloczyn wektorowy

#include <bits/stdc++.h>
using namespace std;

struct Point 
{
    int x, y;    
};

double cross(Point a, Point b)
{
    return a.x * b.y - a.y * b.x;
}

double polygonField(vector<Point>& points)
{
    int n = points.size();
    double field = 0;
    for (int i = 0; i < n - 1; ++i) 
    {
        field += cross(points[i + 1], points[i]) / 2;
    }
    field += cross(points[0], points[n - 1]) / 2;
    return field;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<Point> points(n);
    for (int i = 0; i < n; ++i)
    {
        Point p;
        cin >> p.x >> p.y;
        points[i] = p;
    }

    double field = polygonField(points);
    cout << setprecision(1) << fixed;
    cout << field << '\n';

    return 0;
}
