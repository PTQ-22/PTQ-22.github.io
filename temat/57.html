Najdłuższy podciąg rosnący

#include <bits/stdc++.h>
using namespace std;

int lic(vector<int> &arr)
{
    int n = arr.size();
    vector<int> dp(n, 1);
    
    for (int i = 0; i < n; ++i)
        for (int j = 0; j < i; ++j)
            if (arr[i] > arr[j] && dp[i] < dp[j] + 1)
                dp[i] = dp[j] + 1;

    int maxi = -1;
    for (int i = 0; i < n; ++i)
    {
        maxi = max(maxi, dp[i]);
    }
    return maxi;
}

int main()
{
    fstream input;
    input.open("ciagi.txt");
    ofstream output;
    output.open("podciagi.txt");

    int k;
    input >> k;

    while (k--)
    {
        int n;
        input >> n;
        vector<int> arr(n);
        for (int i = 0; i < n; ++i)
        {
            input >> arr[i];
        }
        output << lic(arr) << '\n';
    }

    input.close();
    output.close();
    return 0;
}
