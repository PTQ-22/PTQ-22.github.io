Najdłuższy Wspólny Podciąg - dynamicznie

#include <bits/stdc++.h>
using namespace std;

string lcsDP(string s1, string s2)
{
    int n = s1.size();
    int m = s2.size();
    vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
    for (int i = 1; i <= n; ++i)
    {
        for (int j = 1; j <= m; ++j)
        {
            if (s1[i - 1] == s2[j - 1])
            {
                dp[i][j] = dp[i - 1][j - 1] + 1;
            }
            else {
                dp[i][j] = max(dp[i - 1][j], dp[j - 1][i]);
            }
        }
    }
    string lcs = "";
    int i = n;
    int j = m;
    while (i > 0 && j > 0)
    {
        if(s1[i - 1] == s2[j - 1])
        {
            lcs = s1[i - 1] + lcs;
            i--;
            j--;
        }
        else if (dp[i - 1][j] > dp[i][j - 1])
        {
            i--;
        }
        else {
            j--;
        }
    }
    return lcs;
}

int main()
{
    string firstText = "abacabbd";
    string secondText = "xaadcbdd";

    cout << lcsDP(firstText, secondText) << '\n';

    return 0;
}
