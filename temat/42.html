Czy dwa punkty leżą po tej samej stronie prostej

#include <bits/stdc++.h>
using namespace std;

struct Point
{
    int x, y;
};

int det(Point a, Point b, Point c)
{
    // | ax ay 1 | ax ay
    // | bx by 1 | bx by
    // | cx cy 1 | cx cy
    return a.x * b.y + a.y * c.x + b.x * c.y 
         - b.y * c.x - a.x * c.y - a.y * b.x;
}

int sgn(int x)
{
    if (x > 0) return 1;
    else if (x == 0) return 0;
    return -1;
}

int main()
{
    Point p, q, a, d;

    cout << "Podaj współrzędne prostej pq: \n";
    cout << "p: (x, y) \n";
    cin >> p.x >> p.y;
    cout << "q: (x, y) \n";
    cin >> q.x >> q.y;


    cout << "Podaj 2 punkty\n";
    cout << "a: (x, y) \n";
    cin >> a.x >> a.y;
    cout << "d: (x, y) \n";
    cin >> d.x >> d.y;

    if (sgn(det(p, q, a)) == sgn(det(p, q, d))) {
        cout << "Punkty a i d LEŻĄ po tej samej stronie prostej pq\n";
    }
    else {
        cout << "Punkty a i d NIE LEŻĄ po tej samej stronie prostej pq\n";
    }

    return 0;
}
