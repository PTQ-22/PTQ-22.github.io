Algorytm stacji rozrządowej Dijkstry

Algorytm stacji rozrządowej Dijkstry - algorytm zamiany notacji infiksowej (nawiasowej) na postfiksową (ONP). 

#include <iostream>
#include <map>
#include <stack>
using namespace std;

string infix_to_ONP(string input)
{
    map<char, int> op_priority = {
        {'(', 0}, {'+', 1}, {'-', 1}, {'*', 2}, {'/', 2}
    };

    stack<char> op_stack;
    string output = "";
  
    for (char c : input) {
        if (c >= '0') { // is number
            output += c;
        }
        else { // is operator
            if (op_stack.empty()) {
                op_stack.push(c);
            }
            else if (c == ')') {
                while (op_stack.top() != '(') {
                    output += op_stack.top();
                    op_stack.pop();
                }
                op_stack.pop(); // remove '('
            }
            else if (op_priority[c] >= op_priority[op_stack.top()] || c == '(') {
                op_stack.push(c);                
            }
            else {
                while (op_priority[c] < op_priority[op_stack.top()]) {
                    output += op_stack.top();
                    op_stack.pop();
                    if (op_stack.empty()) break;
                }
                op_stack.push(c);
            }
        }
    }
    while (!op_stack.empty()) {
        output += op_stack.top();
        op_stack.pop();
    }

    return output;
}

int main()
{
    cout << infix_to_ONP("5*8-6/(1+2)");

    return 0;
}
