Algorytm stacji rozrządowej Dijkstry

Algorytm stacji rozrządowej Dijkstry - algorytm zamiany notacji infiksowej (nawiasowej) na postfiksową (ONP). 

map<char, int> priority = {
    {'(', 0}, {'+', 1}, {'-', 1}, {'*', 2}, {'/', 2}
};

string infixToPostfix(string infix)
{
    stack<char> op_stack;
    string postfix;
    for (char c : infix) {
        if (c >= '0') postfix += c;
        else {
            if (op_stack.empty()) {
                op_stack.push(c);
            }
            else if (c == ')') {
                while (op_stack.top() != '(') {
                    postfix += op_stack.top();
                    op_stack.pop();
                }
                op_stack.pop();
            }
            else if (priority[c] > priority[op_stack.top()] || c == '(') {
                op_stack.push(c);
            }
            else {
                while (op_stack.size() && priority[c] <= priority[op_stack.top()]) {
                    postfix += op_stack.top();
                    op_stack.pop();
                }
                op_stack.push(c);
            }
        }
    }
    while (op_stack.size()) {
        postfix += op_stack.top();
        op_stack.pop();
    }
    return postfix;
}
