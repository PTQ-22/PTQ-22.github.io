#include <iostream>
#include <vector>
#include <fstream>
using namespace std;

constexpr int MAXN = 100'000;
vector<int> G[MAXN];

int graph_deg(int n)
{
    int g_deg = 0;

    for (int i = 1; i <= n; ++i) {
        int v_deg = 0;
        for (int v : G[i]) {
            if (v == i) v_deg += 2;
            else v_deg++;
        }
        g_deg = max(g_deg, v_deg);
    }

    return g_deg;
}

int main()
{
    fstream file;
    file.open("graf0.txt");

    int n, m;
    file >> n >> m;
    int a, b;
    for (int i = 0; i < m; ++i) {
        file >> a >> b;
        G[a].push_back(b);
        G[b].push_back(a);
    }
    file.close();

    cout << graph_deg(n) << '\n';

    return 0;
}
