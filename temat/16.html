Kolejka

Kolejka - dynamiczna struktura danych, w której mamy dostęp tylko do pierwszego elementu znajdującego się na początku oraz możemy dodawać elementy na koniec. FIFO - first in, first out.

struct Element {
    int value;
    Element* behind;
};

Element* first = nullptr;
Element* last = nullptr;

void push(int x) {
    Element* tmp;
    tmp = new Element;
    tmp->value = x;
    tmp->behind = nullptr;

    if (first == nullptr) {
        first = tmp;
        last = tmp;
    }
    else {
        last->behind = tmp;
        last = tmp;
    }
}

int pop() {
    int res = first->value;
    Element* tmp = first;
    first = tmp->behind;
    // usuwanie elementu podobnie jak w stosie
    delete tmp; 
    return res;
}

bool isEmpty() {
    if (first == nullptr) {
        return true;
    }
    return false;
}
